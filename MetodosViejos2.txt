private void setDatos() {
    tabla1.setRowCount(0); // Limpiar la tabla

    // Mostrar las citas en la tabla
    Cola<String[]> colaTemporal = new Cola<>(); // Cola temporal para mantener el orden original
    while (!citasUsuario.estaVacia()) {
        String[] cita = citasUsuario.peek(); // Obtener el primer elemento de la cola
        if (cita.length >= 5) { // Verificar que haya al menos 5 campos en la cita
            String[] datosMostrar = {cita[0], cita[1], cita[4]}; // Nombre, Apellido(s), Ticket
            tabla1.addRow(datosMostrar);
        }
        colaTemporal.enqueue(citasUsuario.dequeue()); // Mover la cita a la cola temporal
    }
